%extend idbattr_info_t
{
  idbattr_info_t(
          const char *name,
          uintptr_t offset,
          size_t width,
          uint64 bitmask=0,
          uchar tag=0,
          uint idi_flags=0)
  {
    idbattr_info_t *ii = new idbattr_info_t();
    ii->name = nullptr;
    if ( name != nullptr )
    {
      // mimick SWiG's `new`-based string allocation
      size_t len = strlen(name) + 1;
      ii->name = (char *) memcpy(new char[len], name, len);
    }
    ii->offset = offset;
    ii->width = width;
    ii->bitmask = bitmask;
    ii->tag = tag;
    ii->idi_flags = idi_flags;
    return ii;
  }

  ~idbattr_info_t()
  {
    delete [] $self->name;
    $self->name = nullptr;
    delete $self;
  }
};
