  }

  {
    $result = PyLong_AsLong(result);
  }
  else
  {
    Swig::DirectorTypeMismatchException::raise(
            SWIG_ErrorType(SWIG_TypeError),
            "in output value of type '" "int" "'" " in method '$symname'");
  }
}
